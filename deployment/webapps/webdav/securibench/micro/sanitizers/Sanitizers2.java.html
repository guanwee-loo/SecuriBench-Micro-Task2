15 /** 
16  *  @servlet description="simple sanitization check" 
17  *  @servlet vuln_count = "0" 
18  *  */
19 public class Sanitizers2 extends HttpServlet implements MicroTestCase {
20     private static final String FIELD_NAME = "name";
21     private PrintWriter writer;
22 
23     protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws IOException {
24         String name = req.getParameter(FIELD_NAME);
25         String clean = clean(name);
26         
27         writer = resp.getWriter();
28         resp.setContentType("text/html");
29         
30         writer.println("<html>" + clean + "</html>");                  /* OK */
31         
32     }
33     
34     /**
35      * @sanitizer 
36      * javascript sanitization routine 
37      * */
38     private String clean(String name) {
39         StringBuffer buf = new StringBuffer();
40         for(int i = 0; i < name.length(); i++) {
41             char ch = name.charAt(i);
42             switch (ch) {
43                 case '<':
44                     buf.append("&lt;"); break;
45                 case '>':
46                     buf.append("&gt;"); break;
47                 case '&':
48                     buf.append("&amp;"); break;
49                 default:
50                     if(Character.isLetter(ch) || Character.isDigit(ch) || ch == '_') {
51                         buf.append(ch);
52                     } else {
53                         buf.append('?');
54                     }
55             }
56         }
57         
58         return buf.toString();
59     }
60 
61     public String getDescription() {
62         return "simple sanitization check";
63     }
64     
65     public int getVulnerabilityCount() {
66         return 1;
67     }
